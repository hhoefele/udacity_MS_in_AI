1. Creating and Calling a Function

def greet(name="Default Name", time="morning"):
    return f"Good {time}, {name}!"

# Calling the function with only the name parameter

# Calling the function with only the time parameter

2. Function with Parameters and Return Values
# Create a function that takes two parameters
# to calculate the are of a rectangle and returns the result
# add optional parameter for default units
def calculate_area(length, width, unit="square meters"):
    area = length * width
    return area, unit

#calculate and store the result in variable
area, unit = calculate_area(5, 10)

#print result using f string
print(f"The area of the rectangle is {area} {unit}.")
